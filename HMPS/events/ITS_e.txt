country_event = {
	id = 4609326
	title = "Replace Cores"
	desc = "Replace Northern Italian cores with Southern Italian Cores."
	
	trigger = {
		tag = ITS
		ITA = { exists = no }
		has_global_flag = ITS_form
		any_core = {
			is_core = ITA
		}
	}
	
	option = {
		name = "Ok"
		ITA = { all_core = { add_core = ITS } }
		ITA = { all_core = { remove_core = ITA } }
	}
}
